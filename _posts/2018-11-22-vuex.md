npm install `vuex` --save （--save生产环境局部安装）

###  store.js

```javascript

import Vue from 'vue';
import Vuex from 'vuex';
Vue.use(Vuex)

 //保存状态
const state={
    role:'',
    todos: [
        { id: 1, text: 'ceshi1 ', done: true },
        { id: 2, text: 'ceshi2 ', done: true },
        { id: 3, text: 'ceshi3 ', done: true },
        { id: 4, text: 'ceshi4', done: false }
    ]
}
 //修改状态
const mutations={
  setRole(state,val){ 
    state.role=val;
  },
}

//计算过滤操作
//就像计算属性一样，getter 的返回值会根据它的依赖被缓存起来，且只有当它的依赖值发生了改变才会被重新计算
const getters = { 
  /*通过属性访问*/
  doneTodos: state => {
     return state.todos.filter(todo => todo.done)
  },
  //两个参数时
  doneTodosCount: (state, getters) => {
      return getters.doneTodos.length
  },
  
  
  /*通过方法访问*/
  //注意，getter 在通过方法访问时，每次都会去进行调用，而不会缓存结果
  getTodoById: (state) => (id) => {
      return state.todos.find(todo => todo.id === id)
  }
}
//actions异步修改状态
const actions ={

}

//将方法暴露出去
export default new Vuex.Store({
  state,mutations,getters,actions
})

```

### state 使用

```html
<p>{{$store.state.role}}</p>
```
```javascript
//1.直接使用
this.$store.state.role;

//2.简化引入
import {mapState} from 'vuex';
computed:mapState(["role",...]);
// 和其他计算属性方法混合使用
computed:{
    ...mapState(["role",...]),
    calc(){} 
 }
```

### mutations使用
```html
<p @click="$store.commit('setRole',val)"></p>
```
```javascript
//1.直接使用
this.$store.commit('setRole',val);

//2.简化引入
import {mapMutations} from 'vuex';
methods:mapMutations(["setRole",...]);
// 和其他计算属性方法混合使用
methods:{
    ...mapMutations(["role",...]),
    calc(){} 
 }
```
### getters

```html
<p>
     <!--通过属性访问-->
    {{$store.getters.doneTodos[0].text}}
    {{$store.getters.doneTodosCount}}
    
    <!--通过方法访问 （传参） 不会缓存-->
    {{$store.getters.getTodoById(2)}}
</p>
```
```javascript
//1.直接使用
this.$store.getters.doneTodos;

//2.简化引入
import {mapGetters} from 'vuex';
computed:mapGetters(["doneTodos",...]);
// 和其他计算属性方法混合使用
computed:{
    ...mapGetters(["doneTodos",...]),
    calc(){} 
 }
```
